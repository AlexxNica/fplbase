183  /// @brief Compute normals and tangents given position and texcoords.
184  ///
185  /// The template type should be a struct with at least the following fields:
186  /// mathfu::vec3_packed pos;
187  /// mathfu::vec2_packed tc;
188  /// mathfu::vec3_packed norm;
189  /// mathfu::vec4_packed tangent;
190  ///
191  /// @param vertices The vertices to computes the information for.
192  /// @param indices The indices that make up the mesh.
193  /// @param numverts The number of vertices in the vertex array.
194  /// @param numindices The number of indices in the index array.
195  template <typename T>
196  static void ComputeNormalsTangents(T *vertices, const unsigned short *indices,
197  int numverts, int numindices) {
198  std::unique_ptr<mathfu::vec3[]> binormals(new mathfu::vec3[numverts]);
199 
200  // set all normals to 0, as we'll accumulate
201  for (int i = 0; i < numverts; i++) {
202  vertices[i].norm = mathfu::kZeros3f;
203  vertices[i].tangent = mathfu::kZeros4f;
204  binormals[i] = mathfu::kZeros3f;
205  }
206  // Go through each triangle and calculate tangent space for it, then
207  // contribute results to adjacent triangles.
208  // For a description of the math see e.g.:
209  // http://www.terathon.com/code/tangent.html
210  for (int i = 0; i < numindices; i += 3) {
211  auto &v0 = vertices[indices[i + 0]];
212  auto &v1 = vertices[indices[i + 1]];
213  auto &v2 = vertices[indices[i + 2]];
214  // The cross product of two vectors along the triangle surface from the
215  // first vertex gives us this triangle's normal.
216  auto q1 = mathfu::vec3(v1.pos) - mathfu::vec3(v0.pos);
217  auto q2 = mathfu::vec3(v2.pos) - mathfu::vec3(v0.pos);
218  auto norm = normalize(cross(q1, q2));
219  // Contribute the triangle normal into all 3 verts:
220  v0.norm = mathfu::vec3(v0.norm) + norm;
221  v1.norm = mathfu::vec3(v1.norm) + norm;
222  v2.norm = mathfu::vec3(v2.norm) + norm;
223  // Similarly create uv space vectors:
224  auto uv1 = mathfu::vec2(v1.tc) - mathfu::vec2(v0.tc);
225  auto uv2 = mathfu::vec2(v2.tc) - mathfu::vec2(v0.tc);
226  float m = 1 / (uv1.x() * uv2.y() - uv2.x() * uv1.y());
227  auto tangent = mathfu::vec4((uv2.y() * q1 - uv1.y() * q2) * m, 0);
228  auto binorm = (uv1.x() * q2 - uv2.x() * q1) * m;
229  v0.tangent = mathfu::vec4(v0.tangent) + tangent;
230  v1.tangent = mathfu::vec4(v1.tangent) + tangent;
231  v2.tangent = mathfu::vec4(v2.tangent) + tangent;
232  binormals[indices[i + 0]] = binorm;
233  binormals[indices[i + 1]] = binorm;
234  binormals[indices[i + 2]] = binorm;
235  }
236  // Normalize per vertex tangent space constributions, and pack tangent /
237  // binormal into a 4 component tangent.
238  for (int i = 0; i < numverts; i++) {
239  auto norm = mathfu::vec3(vertices[i].norm);
240  auto tangent = mathfu::vec4(vertices[i].tangent);
241  // Renormalize all 3 axes:
242  norm = normalize(norm);
243  tangent = mathfu::vec4(normalize(tangent.xyz()), 0);
244  binormals[i] = normalize(binormals[i]);
245  tangent = mathfu::vec4(
246  // Gram-Schmidt orthogonalize xyz components:
247  normalize(tangent.xyz() - norm * dot(norm, tangent.xyz())),
248  // The w component is the handedness, set as difference between the
249  // binormal we computed from the texture coordinates and that from the
250  // cross-product:
251  dot(cross(norm, tangent.xyz()), binormals[i]));
252  vertices[i].norm = norm;
253  vertices[i].tangent = tangent;
254  }
255  }
