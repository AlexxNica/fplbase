69  /// @brief Raw call to set any uniform (with 1/2/3/4/16 components).

79  /// @brief Set an non-standard uniform to a vec2/3/4 value.
80  ///
81  /// Call this after Set() or FindUniform().
82  ///
83  /// @param uniform_loc Handle to the uniform that will be set.
84  /// @param value The vector to set the uniform to.
85  template <int N>
86  void SetUniform(UniformHandle uniform_loc,
87  const mathfu::Vector<float, N> &value) {
88  SetUniform(uniform_loc, &value[0], N);
89  }
90 
91  /// @brief Convenience call that does a Lookup and a Set if found.
92  ///
93  /// Call this after Set().
94  ///
95  /// @param uniform_name The name of the uniform that will be set.
96  /// @param value The vector to set the uniform to.
97  /// @return Returns true if the uniform was found and set, false otherwise.
98  template <int N>
99  bool SetUniform(const char *uniform_name,
100  const mathfu::Vector<float, N> &value) {
101  auto loc = FindUniform(uniform_name);
102  if (loc < 0) return false;
103  SetUniform(loc, &value[0], N);
104  return true;
105  }
106 
107  /// @brief Set a non-standard uniform to a float value.
108  ///
109  /// Call this after Set().
110  ///
111  /// @param uniform_name The name of the uniform that will be set.
112  /// @param value The float to set the uniform to.
113  /// @return Returns true if the uniform was found and set, false otherwise.
114  bool SetUniform(const char *uniform_name, float value) {
115  auto loc = FindUniform(uniform_name);
116  if (loc < 0) return false;
117  SetUniform(loc, &value, 1);
118  return true;
119  }
120 
121  /// @brief Set a non-standard uniform to a mat4 value.
122  ///
123  /// Call this after Set().
124  ///
125  /// @param uniform_name The name of the uniform that will be set.
126  /// @param value The mat4 to set the uniform to.
127  /// @return Returns true if the uniform was found and set, false otherwise.
128  bool SetUniform(const char *uniform_name, const mathfu::mat4 &value) {
129  auto loc = FindUniform(uniform_name);
130  if (loc < 0) return false;
131  SetUniform(loc, &value[0], sizeof(value) / sizeof(float));
132  return true;
133  }
