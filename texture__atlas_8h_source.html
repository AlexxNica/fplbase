31 /// @class TextureAtlas
32 /// @brief Texture coordinate dictionary.
33 ///
34 /// A TextureAtlas supports sprite sheets by containing one texture and a list
35 /// of subtexture bounding boxes, which can also be indexed by name using the
36 /// index_map.
37 ///
38 /// @warning This is will very likely be refactored.
39 class TextureAtlas {
40  public:
41  TextureAtlas() : atlas_texture_(nullptr) {}
42  ~TextureAtlas() { Delete(); }
43 
44  /// @brief Delete the texture associated with this atlas.
45  void Delete() {
46  atlas_texture_->Delete();
47  atlas_texture_ = nullptr;
48  }
49 
50  /// @brief Get the bounds of a subtexture associated with name.
51  ///
52  /// @param name Name of the subtexture to lookup.
53  /// @returns Bounds of the subtexture or nullptr if the specified name isn't
54  /// found.
55  const vec4i *GetBounds(const std::string &name) {
56  auto index_iter = index_map_.find(name);
57  if (index_iter != index_map_.end()) {
58  return &subtexture_bounds_[index_iter->second];
59  }
60  return nullptr;
61  }
62 
63  /// @brief Get the texture associated with this atlas.
64  /// @return Pointer to the texture associated with this atlas.
65  const Texture *atlas_texture() const { return atlas_texture_; }
66  /// @brief Get the texture associated with this atlas.
67  /// @return Pointer to the texture associated with this atlas.
68  Texture *atlas_texture() { return atlas_texture_; }
69  /// @brief Set the texture associated with this atlas.
70  void set_atlas_texture(Texture *atlas_texture) {
71  atlas_texture_ = atlas_texture;
72  if (atlas_texture && atlas_texture->id()) {
73  set_size(atlas_texture->size());
74  }
75  }
76 
77  /// @brief Get the size of the overall texture.
78  /// @return Size of the overall texture.
79  const vec2i &size() const { return size_; }
80  /// @brief Set the size of the overall texture.
81  void set_size(const vec2i &size) { size_ = size; }
82 
83  /// @brief Get a vector of the bounds of each subtexture in this atlas.
84  ///
85  /// Each element of the vector consists of (offsetx, offsety, sizex, sizey)
86  /// where offsetx / offsety are coordinates relative to the underlying
87  /// atlased texture and sizex / sizey are the dimensions of the subtexture.
88  ///
89  /// @returns Vector of subtexture bounds.
90  const std::vector<vec4i> &subtexture_bounds() const {
91  return subtexture_bounds_;
92  }
93  std::vector<vec4i> &subtexture_bounds() { return subtexture_bounds_; }
94 
95  /// @brief Get a map of subtexture names to subtexture offsets.
96  ///
97  /// Each entry in the map can be used to lookup the subtexture bounds in
98  /// the vector returned by @ref subtexture_bounds().
99  ///
100  /// @return A map of subtexture names to indices in vector returned by
101  /// @ref subtexture_bounds().
102  const std::map<std::string, size_t> &index_map() const { return index_map_; }
103  std::map<std::string, size_t> &index_map() { return index_map_; }
104 
105  private:
106  // Texture being used by this atlas.
107  Texture *atlas_texture_;
108  // Size of the texture/
109  vec2i size_;
110  // List of bounds (offsetx, offsety, sizex, sizey) of each subtexture.
111  std::vector<vec4i> subtexture_bounds_;
112  // Map of subtexture names to indices into subtexture_bounds_.
113  std::map<std::string, size_t> index_map_;
114 };
115
