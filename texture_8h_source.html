32 enum TextureFormat {
33  kFormatAuto = 0, ///< @brief The default, picks based on loaded data.

36  kFormat5551,
37  kFormat565,

40  kFormatPKM,
41  kFormatKTX,
42  kFormatCount // Must be at end.

45 /// @brief determines if the format has an alpha component.
46 inline bool HasAlpha(TextureFormat format) {
47  switch (format) {
48  case kFormat8888:
49  case kFormat5551:
50  case kFormatASTC:
51  case kFormatKTX: // TODO(wvo): depends on the internal format.
52  return true;
53  default:
54  return false;
55  }
56 }

58 /// @brief determines if the format is already compressed in some way.
59 /// If image data is supplied in these formats, we use them as-is.
60 inline bool IsCompressed(TextureFormat format) {
61  switch (format) {
62  case kFormat5551:
63  case kFormat565:
64  case kFormatASTC:
65  case kFormatPKM:
66  case kFormatKTX:
67  return true;
68  default:
69  return false;
70  }
71 }
