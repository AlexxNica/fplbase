58 /// @class Button
59 /// @brief Used to record state for fingers, mousebuttons, keys and gamepad
60 /// buttons.
61 ///
62 /// Allows you to know if a button went up/down this frame.
63 class Button {
64  public:
65  Button() : is_down_(false) { AdvanceFrame(); }
66  /// @brief Advances the current state of the button by one frame.
67  ///
68  /// Important, because it tells the system where frame boundaries occur, so
69  /// that went_down() and went_up() can be updated correctly.
70  /// Normally called automatically by FPLBase.
71  void AdvanceFrame() { went_down_ = went_up_ = false; }
72 
73  /// @brief Updates the current state of the button.
74  ///
75  /// For buttons that are tracked by the input system (Keyboard buttons,
76  /// joysticks) this is invoked automatically by InputSystem::AdvanceFrame().)
77  void Update(bool down);
78 
79  /// @brief Returns true if the button is currently pressed.
80  bool is_down() const { return is_down_; }
81 
82  /// @brief Returns true if the button has been pressed since last update.
83  bool went_down() const { return went_down_; }
84 
85  /// @brief Returns true if the button has been released since last update.
86  bool went_up() const { return went_up_; }
87 
88  private:
89  bool is_down_;
90  bool went_down_;
91  bool went_up_;
92 };

117 /// @class InputPointer
118 /// @brief Stores information about the current and recent state of a pointer.

121 /// or a mouse-pointer.

182  void SetHat(size_t hat_index, const mathfu::vec2 &hat);

416 // An event parameters for a keyboard input.

419 // caret.

421  bool state; // key state, true:pressed, false:released.
422  bool repeat; // A flag indicates if the key is repeated input.
423  FPL_Keycode symbol; // Key symbol, refer keyboard_keycodes.h for a
424  // detail.
425  FPL_Keymod modifier; // Modifier key state, refer keyboard_keycodes.h for
426  // a detail.
427 };
