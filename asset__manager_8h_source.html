295  // This implements the mechanism for each asset to be both loadable
296  // sync or async.
297  // It gets passed a blank asset that we take ownership of, and the map it
298  // should go into if all succeeds.
299  template<typename T> T *LoadOrQueue(T *asset,
300  std::map<std::string, T *> &asset_map,
301  bool async) {
302  if (async) {
303  loader_.QueueJob(asset);
304  } else {
305  bool ok = asset->LoadNow();
306  if (!ok) {
307  delete asset;
308  return nullptr;
309  }
310  }
311  asset_map[asset->filename()] = asset;
312  return asset;
313  }
