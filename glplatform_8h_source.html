129 // Define a GL_CALL macro to wrap each (void-returning) OpenGL call.
130 // This logs GL error when LOG_GL_ERRORS below is defined.
131 #if defined(_DEBUG) || DEBUG == 1
132 #define LOG_GL_ERRORS
133 #endif
134 
135 #ifdef LOG_GL_ERRORS
136 #define GL_CALL(call) \
137  { \
138  call; \
139  LogGLError(__FILE__, __LINE__, #call); \
140  }
141 #else
142 #define GL_CALL(call) \
143  do { \
144  call; \
145  } while (0)
146 #endif
147 
148 // The error checking function used by the GL_CALL macro above,
149 // uses glGetError() to check for errors.
150 extern void LogGLError(const char *file, int line, const char *call);
