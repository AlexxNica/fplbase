39 /// @class AsyncResource
40 /// @brief Any resource intended to be loaded asynchronously should inherit from
41 /// this.
42 class AsyncResource {
43  public:
44  AsyncResource() : data_(nullptr) {}
45  explicit AsyncResource(const char *filename)
46  : filename_(filename), data_(nullptr) {}
47  virtual ~AsyncResource() {}
48 
49  /// @brief Override with the actual loading behavior.
50  ///
51  /// Load should perform the actual loading of filename_, and store the
52  /// result in data_, or nullptr upon failure. It is called on the loader
53  /// thread, so should not access any program state outside of this object.
54  /// Since there will be only one loader thread, any libraries called by Load
55  /// need not be MT-safe as long as they're not also called by the main thread.
56  virtual void Load() = 0;
57 
58  /// @brief Override with converting the data into the resource.
59  ///
60  /// This should implement the behavior of turning data_ into the actual
61  /// desired resource. Called on the main thread only.
62  virtual void Finalize() = 0;
63 
64  /// @brief Sets the filename that should be loaded.
65  ///
66  /// Set the the filename in situations where it can't be initialized in
67  /// the constructor. Must be called before AsyncLoader::QueueJob().
68  ///
69  /// @param filename The name of the file to load.
70  void set_filename(const std::string &filename) { filename_ = filename; }
71 
72  /// @brief The name of the file associated with the resource.
73  ///
74  /// @return Returns the filename.
75  const std::string &filename() const { return filename_; }
76 
77  protected:
78  std::string filename_;
79  uint8_t *data_;
80 
81  friend class AsyncLoader;
82 };

84 /// @class AsyncLoader
85 /// @brief Handles loading AsyncResource objects.
86 class AsyncLoader {
87  public:
88  AsyncLoader();
89  ~AsyncLoader();
90 
91  /// @brief Queues AsyncResources to be loaded by StartLoading.
92  ///
93  /// Call this any number of times before StartLoading.
94  ///
95  /// @param res The resource to queue for loading.
96  void QueueJob(AsyncResource *res);
97 
98  /// @brief Launches the loading thread for the previously queued jobs.
99  void StartLoading();
100 
101  /// @brief Ends the loading thread when all jobs are done.
102  ///
103  /// Cleans-up the background loading thread once all jobs have been completed.
104  /// You can restart with StartLoading() if you like.
105  void StopLoadingWhenComplete();
106 
107  /// @brief Call to Finalize any resources that have finished loading.
108  ///
109  /// Call this once per frame after StartLoading. Will call Finalize on any
110  /// resources that have finished loading. One it returns true, that means
111  /// the queue is empty, all resources have been processed, and the loading
112  /// thread has terminated.
113  ///
114  /// @return Returns true once the queue is empty.
115  bool TryFinalize();
116 
117  private:
118 #ifdef FPL_BASE_BACKEND_SDL
119  void Lock(const std::function<void()> &body);
120  template <typename T>
121  T LockReturn(const std::function<T()> &body) {
122  T ret;
123  Lock([&ret, &body]() { ret = body(); });
124  return ret;
125  }
126 #endif
127 
128  void LoaderWorker();
129  static int LoaderThread(void *user_data);
130 
131  std::vector<AsyncResource *> queue_, done_;
132 
133 #ifdef FPL_BASE_BACKEND_SDL
134  // Keep handle to the worker thread around so that we can wait for it to
135  // finish before destroying the class.
136  Thread worker_thread_;
137 
138  // This lock protects ALL state in this class, i.e. the two vectors.
139  Mutex mutex_;
140 
141  // Kick-off the worker thread when a new job arrives.
142  Semaphore job_semaphore_;
143 #elif defined(FPL_BASE_BACKEND_STDLIB)
144  std::thread worker_thread_;
145  std::mutex mutex_;
146  std::condition_variable job_cv_;
147 #else
148 #error Need to define FPL_BASE_BACKEND_XXX
149 #endif
150 };
