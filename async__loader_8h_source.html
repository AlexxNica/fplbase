44 /// @class AsyncResource
45 /// @brief Any resource that can be loaded asynchronously should inherit from
46 /// this.
47 class AsyncAsset : public Asset {
48  public:
49  /// @brief Default constructor for an empty AsyncAsset.
50  AsyncAsset() : data_(nullptr) {}
51 
52  /// @brief Construct an AsyncAsset with a given file name.
53  /// @param[in] filename A C-string corresponding to the name of the asset
54  /// file.
55  explicit AsyncAsset(const char *filename)
56  : filename_(filename), data_(nullptr) {}
57 
58  /// @brief AsyncAsset destructor.
59  virtual ~AsyncAsset() {}
60 
61  /// @brief Override with the actual loading behavior.
62  ///
63  /// Load should perform the actual loading of filename_, and store the
64  /// result in data_, or nullptr upon failure. It is called on the loader
65  /// thread, so should not access any program state outside of this object.
66  /// Since there will be only one loader thread, any libraries called by Load
67  /// need not be MT-safe as long as they're not also called by the main thread.
68  virtual void Load() = 0;
69 
70  /// @brief Override with converting the data into the resource.
71  ///
72  /// This should implement the behavior of turning data_ into the actual
73  /// desired resource. Called on the main thread only.
74  /// Should check if data_ is null.
75  virtual void Finalize() = 0;
76 
77  /// @brief Performs a synchronous load by calling Load & Finalize.
78  ///
79  /// Not used by the loader thread, should be called on the main thread.
80  /// @return Returns false on failure.
81  bool LoadNow() {
82  Load();
83  bool ok = data_ != nullptr;
84  // Call this even if data_ is null, to enforce Finalize() checking for it.
85  Finalize();
86  return ok;
87  }
88 
89  /// @brief Sets the filename that should be loaded.
90  ///
91  /// Set the the filename in situations where it can't be initialized in
92  /// the constructor. Must be called before AsyncLoader::QueueJob().
93  ///
94  /// @param filename The name of the file to load.
95  void set_filename(const std::string &filename) { filename_ = filename; }
96 
97  /// @brief The name of the file associated with the resource.
98  ///
99  /// @return Returns the filename.
100  const std::string &filename() const { return filename_; }
101 
102  protected:
103  /// @brief The resource file name.
104  std::string filename_;
105  /// @brief The resource data.
106  const uint8_t *data_;
107 
108  friend class AsyncLoader;
109 };

111 /// @class AsyncLoader
112 /// @brief Handles loading AsyncAsset objects.
113 class AsyncLoader {
114  public:
115  AsyncLoader();
116  ~AsyncLoader();
117 
118  /// @brief Queues AsyncResources to be loaded by StartLoading.
119  ///
120  /// Call this any number of times before StartLoading.
121  ///
122  /// @param res The resource to queue for loading.
123  void QueueJob(AsyncAsset *res);
124 
125  /// @brief Launches the loading thread for the previously queued jobs.
126  void StartLoading();
127 
128  /// @brief Ends the loading thread when all jobs are done.
129  ///
130  /// Cleans-up the background loading thread once all jobs have been completed.
131  /// You can restart with StartLoading() if you like.
132  void StopLoadingWhenComplete();
133 
134  /// @brief Call to Finalize any resources that have finished loading.
135  ///
136  /// Call this once per frame after StartLoading. Will call Finalize on any
137  /// resources that have finished loading. One it returns true, that means
138  /// the queue is empty, all resources have been processed, and the loading
139  /// thread has terminated.
140  ///
141  /// @return Returns true once the queue is empty.
142  bool TryFinalize();
143 
144  private:
145 #ifdef FPL_BASE_BACKEND_SDL
146  void Lock(const std::function<void()> &body);
147  template <typename T>
148  T LockReturn(const std::function<T()> &body) {
149  T ret;
150  Lock([&ret, &body]() { ret = body(); });
151  return ret;
152  }
153 #endif
154 
155  void LoaderWorker();
156  static int LoaderThread(void *user_data);
157 
158  std::vector<AsyncAsset *> queue_, done_;
159 
160 #ifdef FPL_BASE_BACKEND_SDL
161  // Keep handle to the worker thread around so that we can wait for it to
162  // finish before destroying the class.
163  Thread worker_thread_;
164 
165  // This lock protects ALL state in this class, i.e. the two vectors.
166  Mutex mutex_;
167

171  std::thread worker_thread_;
172  std::mutex mutex_;
173  std::condition_variable job_cv_;
174 #else
175 #error Need to define FPL_BASE_BACKEND_XXX
176 #endif
177 };
