88 /// @brief HighPerformanceParams are used on Android to configure simulated key
89 /// presses, in order to stop the CPU governor (see
90 /// https://www.kernel.org/doc/Documentation/cpu-freq/governors.txt) from
91 /// reducing the CPU frequency when a user is not providing tactile input.
92 ///
93 /// For example, if a user is providing data to the device via a gyroscope, it
94 /// is possible for the CPU governor to reduce the CPU (and potentially the
95 /// GPU) frequency, reducing application performance.
96 struct HighPerformanceParams {
97  HighPerformanceParams()
98  : android_key_code(kDefaultAndroidKeycode),
99  time_between_presses(kDefaultTimeBetweenPresses) {}
100  HighPerformanceParams(int keycode, int presses)
101  : android_key_code(keycode), time_between_presses(presses) {}
102 
103  /// @brief The key to press repeatedly, to keep the CPU active.
104  int android_key_code;
105  // @brief The time (in seconds) between keypress events.
106  double time_between_presses;
107 };
108 #endif
