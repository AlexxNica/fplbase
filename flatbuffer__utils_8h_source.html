34 /// @brief Converts a Vec2.
35 ///
36 /// @param v The Flatbuffer Vec2 to convert.
37 /// @return Returns a converted mathfu vec2.
38 inline mathfu::vec2 LoadVec2(const Vec2* v) {
39  // Note: eschew the constructor that loads contiguous floats. It's faster
40  // than the x, y constructor we use here, but doesn't account for the
41  // endian swap that might occur in Vec3::x().
42  return mathfu::vec2(v->x(), v->y());
43 }

110 /// @brief Converts a Mat3x4 to a mat4.
111 ///
112 /// Affine transform can be serialized as a 3x4 matrix (three rows of four
113 /// elements). The fourth row is (0,0,0,1). We load the rows into the columns
114 /// of a 4x4 matrix, transpose it, and return it as a `mathfu::AffineTransform`.
115 ///
116 /// @param m The Flatbuffer Mat3x4 to convert.
117 /// @return Returns a converted mathfu mat4.
118 inline mathfu::AffineTransform LoadAffine(const Mat3x4* m) {
119  const mathfu::vec4 c0 = LoadVec4(&m->c0());
120  const mathfu::vec4 c1 = LoadVec4(&m->c1());
121  const mathfu::vec4 c2 = LoadVec4(&m->c2());
122  return mathfu::mat4::ToAffineTransform(
123  mathfu::mat4(c0, c1, c2, mathfu::kAxisW4f).Transpose());
124 }
